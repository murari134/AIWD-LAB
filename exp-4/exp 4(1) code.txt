<script>
    function calculateBMI() {
        // Get values from inputs
        const weight = parseFloat(document.getElementById('weight').value);
        const heightCm = parseFloat(document.getElementById('height').value);
        
        const valueDisplay = document.getElementById('bmi-value');
        const statusDisplay = document.getElementById('bmi-status');

        // Validate inputs
        if (weight > 0 && heightCm > 0) {
            // BMI Formula: weight(kg) / [height(m)]^2
            const heightM = heightCm / 100;
            const bmi = (weight / (heightM * heightM)).toFixed(1);

            // Display BMI Value
            valueDisplay.innerHTML = bmi;

            // Determine Body Type based on criteria
            let status = "";
            let colorClass = "";

            if (bmi < 18) {
                status = "Underweight";
                colorClass = "status-underweight";
            } else if (bmi >= 18 && bmi <= 25) {
                status = "Normal Weight";
                colorClass = "status-normal";
            } else if (bmi > 25 && bmi <= 30) {
                status = "Overweight";
                colorClass = "status-overweight";
            } else {
                status = "Obese";
                colorClass = "status-obese";
            }

            // Apply Status and Color
            statusDisplay.innerHTML = status;
            statusDisplay.className = colorClass;

        } else {
            // Reset if inputs are empty or invalid
            valueDisplay.innerHTML = "--";
            statusDisplay.innerHTML = "Enter valid numbers";
            statusDisplay.className = "";
        }
    }
</script>
